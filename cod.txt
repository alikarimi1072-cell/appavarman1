  let html='';
  menuItems.forEach(item=>{
    if(item.children){
      html+=`<div class="menu-item" id="menu-${item.id}">

function toggleSidebar(){
  const sb=document.getElementById('sidebar');
  if(window.innerWidth<=768){
    sb.classList.toggle('mobile-open');
    document.getElementById('mobile-overlay').classList.toggle('show');

  }

// ===== NAVIGATION =====
function navigate(page){
  APP.currentPage=page;
  // Update active states
  document.querySelectorAll('#sidebar-nav a').forEach(a=>{a.classList.remove('active')});
  const activeLink=document.querySelector(`[data-page="${page}"]`);
  if(activeLink){activeLink.classList.add('active');
    const parent=activeLink.closest('.menu-item');
    if(parent&&parent.querySelector('.submenu'))parent.classList.add('open');
  }
  // Close mobile sidebar
  document.getElementById('sidebar').classList.remove('mobile-open');
  document.getElementById('mobile-overlay').classList.remove('show');
  // Render page
  renderPage(page);
}

function renderDataTable(data,columns,options={}){
  const {searchable=true,sortable=true,paginate=true,perPage=10,actions=[],id='dt-'+Date.now(),onSearch,filterHtml=''}=options;
  const state=APP.pageStates[id]||(APP.pageStates[id]={page:1,sort:null,sortDir:'asc',search:'',filter:''});
  
  // Filter
  let filtered=[...data];
  if(state.search){
    const s=state.search.toLowerCase();
    filtered=filtered.filter(row=>columns.some(c=>String(row[c.key]||'').toLowerCase().includes(s)));
  }
  if(options.filterFn&&state.filter)filtered=filtered.filter(r=>options.filterFn(r,state.filter));
  
  // Sort
  if(state.sort){
    const dir=state.sortDir==='asc'?1:-1;
    filtered.sort((a,b)=>{
      const va=a[state.sort],vb=b[state.sort];
      if(typeof va==='number')return(va-vb)*dir;
      return String(va||'').localeCompare(String(vb||''),'fa')*dir;
    });
  }
  
  const total=filtered.length;
  const totalPages=Math.ceil(total/perPage)||1;
  if(state.page>totalPages)state.page=totalPages;
  const start=(state.page-1)*perPage;
  const pageData=paginate?filtered.slice(start,start+perPage):filtered;
  
  let html=`<div class="table-toolbar">`;
  if(searchable)html+=`<div class="table-search"><svg viewBox="0 0 24 24">${ICONS.search}</svg><input type="text" placeholder="جستجو..." value="${state.search}" onkeyup="dtSearch('${id}',this.value)"></div>`;
  html+=filterHtml;
  html+=`<div style="margin-right:auto">${options.toolbarButtons||''}</div></div>`;
  
  html+=`<div class="table-container"><table class="data-table"><thead><tr>`;
  if(options.showRow)html+=`<th>#</th>`;
  columns.forEach(c=>{
    const sortIcon=state.sort===c.key?(state.sortDir==='asc'?'▲':'▼'):'';
    html+=`<th onclick="dtSort('${id}','${c.key}')">${c.label} <span class="sort-icon">${sortIcon}</span></th>`;
  });
  if(actions.length)html+=`<th>عملیات</th>`;
  html+=`</tr></thead><tbody>`;
  
  if(pageData.length===0){
    html+=`<tr><td colspan="${columns.length+(actions.length?1:0)+(options.showRow?1:0)}" class="text-center text-muted" style="padding:2rem">داده‌ای یافت نشد</td></tr>`;
  }
  pageData.forEach((row,i)=>{
    html+=`<tr>`;
    if(options.showRow)html+=`<td>${start+i+1}</td>`;
    columns.forEach(c=>{
      let val=c.render?c.render(row[c.key],row):row[c.key];
      html+=`<td>${val??''}</td>`;
    });
    if(actions.length){
      html+=`<td><div class="actions">`;
      actions.forEach(a=>{
        html+=`<button class="btn btn-xs ${a.class||'btn-outline'}" onclick="${a.handler}(${row.id})" title="${a.label}">${a.icon?icon(a.icon,14):a.label}</button>`;
      });
      html+=`</div></td>`;
    }
    html+=`</tr>`;
  });
  html+=`</tbody></table></div>`;
  
  if(paginate&&totalPages>1){
    html+=`<div style="display:flex;justify-content:space-between;align-items:center;margin-top:.75rem;flex-wrap:wrap;gap:.5rem">
      <span class="text-muted" style="font-size:.78rem">نمایش ${start+1} تا ${Math.min(start+perPage,total)} از ${total} مورد</span>
      <div class="pagination">
        <button ${state.page<=1?'disabled':''} onclick="dtPage('${id}',1)">«</button>
        <button ${state.page<=1?'disabled':''} onclick="dtPage('${id}',${state.page-1})">‹</button>`;
    for(let p=Math.max(1,state.page-2);p<=Math.min(totalPages,state.page+2);p++){
      html+=`<button class="${p===state.page?'active':''}" onclick="dtPage('${id}',${p})">${p}</button>`;
    }
    html+=`<button ${state.page>=totalPages?'disabled':''} onclick="dtPage('${id}',${state.page+1})">›</button>
        <button ${state.page>=totalPages?'disabled':''} onclick="dtPage('${id}',${totalPages})">»</button>
      </div></div>`;
  }
  return html;
}

// ===== CHARTS =====
function drawBarChart(canvasId,labels,datasets,title=''){
  setTimeout(()=>{
    const canvas=document.getElementById(canvasId);if(!canvas)return;
    const ctx=canvas.getContext('2d');
    const W=canvas.parentElement.offsetWidth;const H=300;
    canvas.width=W;canvas.height=H;
    const pad={top:40,right:20,bottom:60,left:70};
    const chartW=W-pad.left-pad.right;const chartH=H-pad.top-pad.bottom;
    const maxVal=Math.max(...datasets.flatMap(d=>d.data))*1.1||100;
    const barGroupW=chartW/labels.length;
    const barW=Math.min(barGroupW*0.6/datasets.length,40);
    
    ctx.clearRect(0,0,W,H);
    ctx.fillStyle='#1e293b';ctx.font='bold 12px Tahoma';ctx.textAlign='center';
    ctx.fillText(title,W/2,20);
    
    // Grid & Y labels
    ctx.strokeStyle='#e2e8f0';ctx.lineWidth=1;ctx.font='10px Tahoma';ctx.textAlign='left';
    for(let i=0;i<=5;i++){
      const y=pad.top+chartH-chartH*i/5;
      ctx.beginPath();ctx.moveTo(pad.left,y);ctx.lineTo(W-pad.right,y);ctx.stroke();
      ctx.fillStyle='#64748b';
      ctx.fillText(formatNum(Math.round(maxVal*i/5)),5,y+4);
    }
    
    // Bars
    const colors=['#4f46e5','#0891b2','#059669','#d97706','#dc2626'];
    datasets.forEach((ds,di)=>{
      ctx.fillStyle=ds.color||colors[di%5];
      ds.data.forEach((val,li)=>{
        const barH=(val/maxVal)*chartH;
        const x=pad.left+li*barGroupW+(barGroupW-barW*datasets.length)/2+di*barW;
        const y=pad.top+chartH-barH;
        ctx.beginPath();
        ctx.roundRect?ctx.roundRect(x,y,barW-2,barH,3):ctx.rect(x,y,barW-2,barH);
        ctx.fill();
      });
    });
    
    // X labels
    ctx.fillStyle='#64748b';ctx.font='10px Tahoma';ctx.textAlign='center';
    labels.forEach((l,i)=>{
      ctx.fillText(l,pad.left+i*barGroupW+barGroupW/2,H-pad.bottom+20);
    });
    
    // Legend
    if(datasets.length>1){
      let lx=W/2-datasets.length*50;
      datasets.forEach((ds,i)=>{
        ctx.fillStyle=ds.color||colors[i%5];
        ctx.fillRect(lx,H-15,10,10);
        ctx.fillStyle='#64748b';ctx.font='10px Tahoma';ctx.textAlign='right';
        ctx.fillText(ds.label,lx-5,H-5);
        lx+=100;
      });
    }
  },50);
}

function drawLineChart(canvasId,labels,datasets,title=''){
  setTimeout(()=>{
    const canvas=document.getElementById(canvasId);if(!canvas)return;
    const ctx=canvas.getContext('2d');
    const W=canvas.parentElement.offsetWidth;const H=300;
    canvas.width=W;canvas.height=H;
    const pad={top:40,right:20,bottom:60,left:70};
    const chartW=W-pad.left-pad.right;const chartH=H-pad.top-pad.bottom;
    const maxVal=Math.max(...datasets.flatMap(d=>d.data))*1.1||100;
    
    ctx.clearRect(0,0,W,H);
    ctx.fillStyle='#1e293b';ctx.font='bold 12px Tahoma';ctx.textAlign='center';
    ctx.fillText(title,W/2,20);
    
    ctx.strokeStyle='#e2e8f0';ctx.lineWidth=1;ctx.font='10px Tahoma';ctx.textAlign='left';
    for(let i=0;i<=5;i++){
      const y=pad.top+chartH-chartH*i/5;
      ctx.beginPath();ctx.moveTo(pad.left,y);ctx.lineTo(W-pad.right,y);ctx.stroke();
      ctx.fillStyle='#64748b';ctx.fillText(formatNum(Math.round(maxVal*i/5)),5,y+4);
    }
    
    const colors=['#4f46e5','#0891b2','#059669','#d97706'];
    datasets.forEach((ds,di)=>{
      const color=ds.color||colors[di%4];
      ctx.strokeStyle=color;ctx.lineWidth=2;ctx.beginPath();
      ds.data.forEach((val,i)=>{
        const x=pad.left+i*(chartW/(labels.length-1||1));
        const y=pad.top+chartH-(val/maxVal)*chartH;
        i===0?ctx.moveTo(x,y):ctx.lineTo(x,y);
      });
      ctx.stroke();
      // Dots
      ctx.fillStyle=color;
      ds.data.forEach((val,i)=>{
        const x=pad.left+i*(chartW/(labels.length-1||1));
        const y=pad.top+chartH-(val/maxVal)*chartH;
        ctx.beginPath();ctx.arc(x,y,4,0,Math.PI*2);ctx.fill();
      });
    });
    
    ctx.fillStyle='#64748b';ctx.font='10px Tahoma';ctx.textAlign='center';
    labels.forEach((l,i)=>{
      const x=pad.left+i*(chartW/(labels.length-1||1));
      ctx.fillText(l,x,H-pad.bottom+20);
    });
  },50);
}

function drawPieChart(canvasId,labels,data,title=''){
  setTimeout(()=>{
    const canvas=document.getElementById(canvasId);if(!canvas)return;
    const ctx=canvas.getContext('2d');
    const W=canvas.parentElement.offsetWidth;const H=300;
    canvas.width=W;canvas.height=H;
    const cx=W/2;const cy=H/2+10;const r=Math.min(W,H)/2-50;
    const total=data.reduce((s,v)=>s+v,0)||1;
    const colors=['#4f46e5','#0891b2','#059669','#d97706','#dc2626','#7c3aed','#db2777','#ea580c','#16a34a','#0284c7'];
    
    ctx.clearRect(0,0,W,H);
    ctx.fillStyle='#1e293b';ctx.font='bold 12px Tahoma';ctx.textAlign='center';
    ctx.fillText(title,W/2,20);
    
    let startAngle=-Math.PI/2;
    data.forEach((val,i)=>{
      const slice=val/total*Math.PI*2;
      ctx.fillStyle=colors[i%colors.length];
      ctx.beginPath();ctx.moveTo(cx,cy);ctx.arc(cx,cy,r,startAngle,startAngle+slice);ctx.closePath();ctx.fill();
      // Label
      const mid=startAngle+slice/2;
      const lx=cx+Math.cos(mid)*(r*0.65);
      const ly=cy+Math.sin(mid)*(r*0.65);
      ctx.fillStyle='#fff';ctx.font='bold 10px Tahoma';ctx.textAlign='center';
      if(val/total>0.05)ctx.fillText(Math.round(val/total*100)+'%',lx,ly);
      startAngle+=slice;
    });
    
    // Legend
    ctx.font='10px Tahoma';ctx.textAlign='right';
    labels.forEach((l,i)=>{
      const lx=W-10;const ly=40+i*18;
      ctx.fillStyle=colors[i%colors.length];ctx.fillRect(lx-8,ly-8,10,10);
      ctx.fillStyle='#64748b';ctx.fillText(l,lx-14,ly);
    });
  },50);
}

function saveCustomer(e,id){

  const data={name:f.get('name'),company:f.get('company'),phone:f.get('phone'),province:f.get('cust_province'),city:f.get('cust_city'),address:f.get('address'),postalCode:f.get('postalCode'),status:f.get('status')};
  if(!data.name||!data.phone){showNotification('لطفاً فیلدهای الزامی را پر کنید','error');return;}
  if(id){const idx=DB.customers.findIndex(c=>c.id===id);if(idx>=0)Object.assign(DB.customers[idx],data);showNotification('مشتری با موفقیت بروزرسانی شد');}
  else{data.id=genId();data.balance=0;data.createdAt=new Date().toLocaleDateString('fa-IR');DB.customers.push(data);showNotification('مشتری جدید با موفقیت اضافه شد');}
  closeModal();renderCustomers();
}

function saveJournalEntry(e){
  e.preventDefault();const f=new FormData(e.target);
  const codes=f.getAll('accountCode[]');const debits=f.getAll('debit[]');const credits=f.getAll('credit[]');
  const lines=codes.map((c,i)=>({accountCode:c,accountName:DB.chartOfAccounts.find(a=>a.code===c)?.name||'',debit:+debits[i],credit:+credits[i]}));
  const totalD=lines.reduce((s,l)=>s+l.debit,0);const totalC=lines.reduce((s,l)=>s+l.credit,0);

  row.querySelector('[name="price[]"]').value=price;
  calcInvTotal();
}

function calcInvTotal(){
  const form=document.getElementById('invoice-form');if(!form)return;
  const qtys=form.querySelectorAll('[name="qty[]"]');
  const prices=form.querySelectorAll('[name="price[]"]');
  const amounts=form.querySelectorAll('[name="amount[]"]');
  let subtotal=0;
  qtys.forEach((q,i)=>{const a=+q.value*(+prices[i].value);amounts[i].value=a;subtotal+=a;});
  const disc=+form.querySelector('[name="discount"]').value;
  const tax=Math.round((subtotal-disc)*0.09);
  form.querySelector('[name="tax"]').value=tax;
  form.querySelector('[name="total"]').value=subtotal-disc+tax;
}

function saveInvoice(e,type,subtype){
  e.preventDefault();const f=new FormData(e.target);
  const isSales=type==='sales';
  const personId=+f.get('personId');
  const person=isSales?DB.customers.find(c=>c.id===personId):DB.suppliers.find(s=>s.id===personId);
  const pIds=f.getAll('productId[]');const qtys=f.getAll('qty[]');const prices=f.getAll('price[]');const amounts=f.getAll('amount[]');
  const items=pIds.map((pid,i)=>({productId:+pid,name:DB.products.find(p=>p.id===+pid)?.name||'',qty:+qtys[i],price:+prices[i],amount:+amounts[i]}));
  const subtotal=items.reduce((s,it)=>s+it.amount,0);
  const inv={id:genId(),number:(isSales?'SI-':'PI-')+(isSales?1400+DB.salesInvoices.length:2400+DB.purchaseInvoices.length),date:f.get('date'),
    [isSales?'customerId':'supplierId']:personId,[isSales?'customerName':'supplierName']:person?.name||'',
    items,subtotal,discount:+f.get('discount'),tax:+f.get('tax'),total:+f.get('total'),status:'unpaid',type:subtype};
  if(isSales)DB.salesInvoices.push(inv);else DB.purchaseInvoices.push(inv);
  closeModal();showNotification('فاکتور با موفقیت ثبت شد');renderInvoices(type,subtype);
}

        <td>${fy.startDate}</td>

        <td>${fy.description||'-'}</td>
        <td><span class="status-badge ${fy.status==='active'?'status-active':'status-inactive'}">${fy.status==='active'?'فعال':'بسته'}</span></td>

}

function showFiscalYearForm(fy=null){
  const isEdit=!!fy;
  const nextYear=fy?fy.year:(DB.fiscalYears&&DB.fiscalYears.length?Math.max(...DB.fiscalYears.map(f=>+f.year))+1:1403);
  showModal(isEdit?'ویرایش سال مالی':'ثبت سال مالی جدید',`
    <form id="fy-form" onsubmit="saveFiscalYear(event,${fy?fy.id:'null'})">
      <div class="form-row">
        <div class="form-group"><label class="form-label">عنوان سال مالی <span class="required">*</span></label><input class="form-control" name="title" value="${fy?.title||'سال مالی '+nextYear}" required></div>
        <div class="form-group"><label class="form-label">سال <span class="required">*</span></label><input type="number" class="form-control" name="year" value="${fy?.year||nextYear}" required></div>
      </div>
      <div class="form-row">
        <div class="form-group"><label class="form-label">تاریخ شروع <span class="required">*</span></label><input class="form-control" name="startDate" value="${fy?.startDate||nextYear+'/01/01'}" placeholder="مثال: 1403/01/01" required></div>
        <div class="form-group"><label class="form-label">تاریخ پایان <span class="required">*</span></label><input class="form-control" name="endDate" value="${fy?.endDate||nextYear+'/12/29'}" placeholder="مثال: 1403/12/29" required></div>
      </div>
      <div class="form-group"><label class="form-label">وضعیت</label><select class="form-control" name="status"><option value="active" ${!fy||fy?.status==='active'?'selected':''}>فعال</option><option value="closed" ${fy?.status==='closed'?'selected':''}>بسته</option></select></div>
      <div class="form-group"><label class="form-label">توضیحات</label><textarea class="form-control" name="description">${fy?.description||''}</textarea></div>
    </form>`,
  `<button class="btn btn-primary" onclick="document.getElementById('fy-form').requestSubmit()">${isEdit?'بروزرسانی':'ثبت سال مالی'}</button><button class="btn btn-outline" onclick="closeModal()">انصراف</button>`);
}

function saveFiscalYear(e,id){
  e.preventDefault();const f=new FormData(e.target);
  if(!DB.fiscalYears)DB.fiscalYears=[];
  const data={title:f.get('title'),year:f.get('year'),startDate:f.get('startDate'),endDate:f.get('endDate'),status:f.get('status'),description:f.get('description')};
  if(!data.title||!data.year||!data.startDate||!data.endDate){showNotification('لطفاً تمام فیلدهای الزامی را پر کنید','error');return;}
  // If setting active, deactivate others
  if(data.status==='active'){DB.fiscalYears.forEach(fy=>fy.status='closed');DB.company.fiscalYear=data.year;}
  if(id){
    const idx=DB.fiscalYears.findIndex(fy=>fy.id===id);
    if(idx>=0)Object.assign(DB.fiscalYears[idx],data);
    showNotification('سال مالی با موفقیت بروزرسانی شد');
  } else {
    data.id=genId();
    DB.fiscalYears.push(data);
    showNotification('سال مالی جدید با موفقیت ثبت شد');
  }
  closeModal();renderFiscalYear();
}

function editFiscalYear(id){
  if(!DB.fiscalYears)return;
  const fy=DB.fiscalYears.find(f=>f.id===id);
  if(fy)showFiscalYearForm(fy);
}

function deleteFiscalYear(id){
  showModal('تأیید حذف','<p>آیا از حذف این سال مالی اطمینان دارید؟ این عملیات قابل بازگشت نیست.</p>',
  `<button class="btn btn-danger" onclick="confirmDeleteFiscalYear(${id})">حذف</button><button class="btn btn-outline" onclick="closeModal()">انصراف</button>`,'modal-sm');
}

function confirmDeleteFiscalYear(id){
  if(!DB.fiscalYears)return;
  const fy=DB.fiscalYears.find(f=>f.id===id);
  if(fy&&fy.status==='active'){showNotification('امکان حذف سال مالی فعال وجود ندارد. ابتدا سال مالی دیگری را فعال کنید.','error');closeModal();return;}
  DB.fiscalYears=DB.fiscalYears.filter(f=>f.id!==id);
  closeModal();showNotification('سال مالی با موفقیت حذف شد');renderFiscalYear();
}

function closeFiscalYear(id){
  showModal('بستن سال مالی','<p>آیا از بستن این سال مالی اطمینان دارید؟ پس از بستن، امکان ثبت سند در این سال وجود نخواهد داشت.</p>',
  `<button class="btn btn-warning" onclick="confirmCloseFiscalYear(${id})">بستن سال مالی</button><button class="btn btn-outline" onclick="closeModal()">انصراف</button>`,'modal-sm');
}

function confirmCloseFiscalYear(id){
  if(!DB.fiscalYears)return;
  const fy=DB.fiscalYears.find(f=>f.id===id);
  if(fy){fy.status='closed';DB.company.fiscalYear='';}
  closeModal();showNotification('سال مالی بسته شد');renderFiscalYear();
}

function activateFiscalYear(id){
  if(!DB.fiscalYears)return;
  DB.fiscalYears.forEach(fy=>fy.status='closed');
  const fy=DB.fiscalYears.find(f=>f.id===id);
  if(fy){fy.status='active';DB.company.fiscalYear=fy.year;}
  showNotification('سال مالی '+fy.year+' فعال شد','success');renderFiscalYear();
}

function createBackup(){
  const data=JSON.stringify(DB,null,2);
  const blob=new Blob([data],{type:'application/json'});
  const url=URL.createObjectURL(blob);
  const a=document.createElement('a');
  a.href=url;a.download='backup_'+new Date().toLocaleDateString('fa-IR').replace(/\//g,'_')+'.json';
  a.click();URL.revokeObjectURL(url);
  showNotification('نسخه پشتیبان با موفقیت ایجاد و دانلود شد','success');
}
